pipeline {
  agent any
    
  tools {nodejs "node"}
    
  stages {
    stage("Clone code from GitHub") {
            steps {
                script {
                    checkout([$class: 'GitSCM', branches: [[name: '*/main']], extensions: [], userRemoteConfigs: [[credentialsId: 'GITHUB-CREDENTIAL', url: 'https://github.com/pratibhak1111/kubernetes_jenkins_pipeline']]])
                }
            }
        }
     
    stage('Node JS Build') {
      steps {
        sh 'npm install'
      }
    }
  
     stage('Build Node JS Docker Image') {
            steps {
                script {
                  sh 'docker build -t pratibhak29/node-app .'
                }
            }
        }


        stage('Deploy Docker Image to DockerHub') {
            steps {
                script {
                 withCredentials([string(credentialsId: 'pratibhak29', variable: 'pratibhak29')]) {
                    sh 'docker login -u pratibhak29 -p ${pratibhak29}'
                 }  
                 sh 'docker push pratibhak29/node-app'
                }
            }
        }
         
     stage('Deploying Node App helm chart on EKS') {
      steps {
        script {
           sh ("aws eks update-kubeconfig --name ekscluster --region us-east-1")
           sh "kubectl get ns"
           sh "helm install nodeapp ./node-app"
        }
       }
      }
    }
}
